def casesbycont(df):
    continents = dict()
    for i, row in df.iterrows():
        cont = row['continent']
        ncases = row['new_cases']
        if cont in continents and cont is float:
            continents[cont] = contintents[cont] + ncases
        else:
            continents[cont] = ncases
    return continents
